# sortable

@short: defines whether sorting on clicking headers of columns is enabled

@signature: {'sortable?: boolean;'}

@default: true

@example:
var treegrid = new dhx.TreeGrid("treegrid_container", {
    columns: [// columns config],
    sortable:false, /*!*/  
    data: dataset
});

@descr:
