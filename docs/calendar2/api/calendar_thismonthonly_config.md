@short: hides dates of the previous/next months relative to the currently displayed one

@signature: {'thisMonthOnly?: boolean;'}

@default: false

@type: boolean

@example:
var calendar = new dhx.Calendar("calendar_container", {
   thisMonthOnly:true
});
