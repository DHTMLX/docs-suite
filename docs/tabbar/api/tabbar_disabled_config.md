# disabled

@short: makes a tab/tabs disabled

@signature: {'disabled?: string | string[];'}

@example:
// make a tab disabled
var tabbar = new dhx.Tabbar("tabbar_container", {
    disabled: "London"
});

// make tabs disabled
var tabbar = new dhx.Tabbar("tabbar_container", {
    disabled: ["London", "Rome"]
});
