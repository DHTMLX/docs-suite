# updateFromResponse

@short:  updates file attributes with data from a server response

@signature: {'updateFromResponse?: boolean;'}

@default: true

@example:

    uploader:{
    	// obligatory, the path for upload
    	target:"/upload"
    	updateFromResponse:false
   	}
